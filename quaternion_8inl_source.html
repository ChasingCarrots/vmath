14 template<typename T>
15 Quaternion<T>::Quaternion(const T pitch, const T yaw, const T roll) {
16  T half_pitch = pitch / static_cast<T>(2.0);
17  T half_yaw = yaw / static_cast<T>(2.0);
18  T half_roll = roll / static_cast<T>(2.0);
19  T sinp = std::sin(half_pitch);
20  T siny = std::sin(half_yaw);
21  T sinr = std::sin(half_roll);
22  T cosp = std::cos(half_pitch);
23  T cosy = std::cos(half_yaw);
24  T cosr = std::cos(half_roll);
25  this->w = cosp * cosy * cosr + sinp * siny * sinr;
26  this->x = sinp * cosy * cosr - cosp * siny * sinr;
27  this->y = cosp * siny * cosr + sinp * cosy * sinr;
28  this->z = cosp * cosy * sinr - sinp * siny * cosr;
29  this->normalize();
30 }

42 template<typename T>
43 Quaternion<T>::Quaternion(const Matrix<T, 4, 4>& mat) {
44  T x = std::sqrt(std::max(static_cast<T>(0.0),
45  static_cast<T>(1.0) + mat[0][0] - mat[1][1] - mat[2][2])) / static_cast<T>(2.0);
46  T y = std::sqrt(std::max(static_cast<T>(0.0),
47  static_cast<T>(1.0) - mat[0][0] + mat[1][1] - mat[2][2])) / static_cast<T>(2.0);
48  T z = std::sqrt(std::max(static_cast<T>(0.0),
49  static_cast<T>(1.0) - mat[0][0] - mat[1][1] + mat[2][2])) / static_cast<T>(2.0);
50  T w = std::sqrt(std::max(static_cast<T>(0.0),
51  static_cast<T>(1.0) + mat[0][0] + mat[1][1] + mat[2][2])) / static_cast<T>(2.0);
52  this->x = std::copysign(x, mat[1][2] - mat[2][1]);
53  this->y = std::copysign(y, mat[2][0] - mat[0][2]);
54  this->z = std::copysign(z, mat[0][1] - mat[1][0]);
55  this->w = w;
56  this->normalize();
57 }

191 Matrix<T, 4> Quaternion<T>::to_matrix() const {
192  Matrix<T, 4, 4> mat;
193  mat[0][0] = static_cast<T>(1.0) - static_cast<T>(2.0) * (this->y * this->y) - static_cast<T>(2.0) * (this->z * this->z);
194  mat[0][1] = static_cast<T>(2.0) * (this->x * this->y) + static_cast<T>(2.0) * (this->w * this->z);
195  mat[0][2] = static_cast<T>(2.0) * (this->x * this->z) - static_cast<T>(2.0) * (this->w * this->y);
196  mat[0][3] = static_cast<T>(0.0);
197  mat[1][0] = static_cast<T>(2.0) * (this->x * this->y) - static_cast<T>(2.0) * (this->w * this->z);
198  mat[1][1] = static_cast<T>(1.0) - static_cast<T>(2.0) * (this->x * this->x) - static_cast<T>(2.0) * (this->z * this->z);
199  mat[1][2] = static_cast<T>(2.0) * (this->y * this->z) + static_cast<T>(2.0) * (this->w * this->x);
200  mat[1][3] = static_cast<T>(0.0);
201  mat[2][0] = static_cast<T>(2.0) * (this->x * this->z) + static_cast<T>(2.0) * (this->w * this->y);
202  mat[2][1] = static_cast<T>(2.0) * (this->y * this->z) - static_cast<T>(2.0) * (this->w * this->x);
203  mat[2][2] = static_cast<T>(1.0) - static_cast<T>(2.0) * (this->x * this->x) - static_cast<T>(2.0) * (this->y * this->y);
204  mat[2][3] = static_cast<T>(0.0);
205  mat[3][0] = static_cast<T>(0.0);
206  mat[3][1] = static_cast<T>(0.0);
207  mat[3][2] = static_cast<T>(0.0);
208  mat[3][3] = static_cast<T>(1.0);
209  return mat;

252 Quaternion<T> Quaternion<T>::slerp(const Quaternion<T>& start, const Quaternion<T>& end, const T t, const bool shortest) {
253  T d = Quaternion<T>::dot(start, end);
254  Quaternion<T> h;
255  if (d < static_cast<T>(0.0) && shortest) {
256  d = -d;
257  h = -end;

259  h = end;

261  T angle = std::acos(d);
262  return (start * std::sin(angle * (static_cast<T>(1.0) - t)) + h * (std::sin(angle * t))) / std::sin(angle);
