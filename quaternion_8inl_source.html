34 template<typename T>
35 quaternion<T>::quaternion(const T& pitch, const T& yaw, const T& roll) {
36  T half_pitch = pitch / static_cast<T>(2.0);
37  T half_yaw = yaw / static_cast<T>(2.0);
38  T half_roll = roll / static_cast<T>(2.0);
39  T sinp = math::sin(half_pitch);
40  T siny = math::sin(half_yaw);
41  T sinr = math::sin(half_roll);
42  T cosp = math::cos(half_pitch);
43  T cosy = math::cos(half_yaw);
44  T cosr = math::cos(half_roll);
45  this->w = cosp * cosy * cosr + sinp * siny * sinr;
46  this->x = sinp * cosy * cosr - cosp * siny * sinr;
47  this->y = cosp * siny * cosr + sinp * cosy * sinr;
48  this->z = cosp * cosy * sinr - sinp * siny * cosr;
49  this->normalize();
50 }

57 quaternion<T>::quaternion(const matrix<T, 4, 4>& mat) {
58  T x = math::sqrt(math::max(static_cast<T>(0.0),
59  static_cast<T>(1.0) + mat[0][0] - mat[1][1] - mat[2][2])) / static_cast<T>(2.0);
60  T y = math::sqrt(math::max(static_cast<T>(0.0),
61  static_cast<T>(1.0) - mat[0][0] + mat[1][1] - mat[2][2])) / static_cast<T>(2.0);
62  T z = math::sqrt(math::max(static_cast<T>(0.0),
63  static_cast<T>(1.0) - mat[0][0] - mat[1][1] + mat[2][2])) / static_cast<T>(2.0);
64  T w = math::sqrt(math::max(static_cast<T>(0.0),
65  static_cast<T>(1.0) + mat[0][0] + mat[1][1] + mat[2][2])) / static_cast<T>(2.0);
66  this->x = math::copysign(x, mat[1][2] - mat[2][1]);
67  this->y = math::copysign(y, mat[2][0] - mat[0][2]);
68  this->z = math::copysign(z, mat[0][1] - mat[1][0]);
69  this->w = w;
70  this->normalize();
71 }

461 quaternion<T> quaternion<T>::slerp(const quaternion<T>& S, const quaternion<T>& E, const T& t, const bool shortest) {
462  T d = quaternion<T>::dot(S, E);
463  quaternion<T> H;
464  if (d < static_cast<T>(0.0) && shortest) {
