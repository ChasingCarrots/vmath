203 template<typename U, typename>
204 T matrix<T, M, N>::minor(const std::size_t row, const std::size_t col) const {
205  matrix<T, M - 1, N - 1> minor_mat;
206  std::size_t i = (col == 0) ? 1 : 0;
207  std::size_t i_minor = 0;
208  while (i_minor < N - 1) {
209  std::size_t j_minor = 0;
210  std::size_t j = (row == 0) ? 1 : 0;
211  while (j_minor < M - 1) {
212  minor_mat[i_minor][j_minor] = this->data[i][j];
213  j_minor++;
214  j = (j + 1 == row) ? (j + 2) : (j + 1);
215  }
216  i_minor++;
217  i = (i + 1 == col) ? (i + 2) : (i + 1);
218  }
219  return minor_mat.det();
220 }
