317 template<typename T, std::size_t N>
318 vector<T, N> vector<T, N>::refract(const vector<T, N>& I, const vector<T, N>& S, const T& eta) {
319  T S_dot_I = vector<T, N>::dot(S, I);
320  T k = static_cast<T>(1.0) - eta * eta * (static_cast<T>(1.0) - S_dot_I * S_dot_I);
321  if (k < static_cast<T>(0.0)) {
322  return vector<T, N>();
323  } else {
324  return I * eta - S * (eta * S_dot_I + math::sqrt(k));
325  }
326 }
