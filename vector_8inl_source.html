155 Vector<T, N> Vector<T, N>::refract(const Vector<T, N>& incident, const Vector<T, N>& surface_normal, const T eta) {
156  T s_dot_i = Vector<T, N>::dot(surface_normal, incident);
157  T k = static_cast<T>(1.0) - eta * eta * (static_cast<T>(1.0) - s_dot_i * s_dot_i);
158  if (k < static_cast<T>(0.0)) {
159  return Vector<T, N>();
160  } else {
161  return incident * eta - surface_normal * (eta * s_dot_i + std::sqrt(k));
162  }
163 }
